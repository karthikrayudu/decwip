var genericTypes = file.Root.DescendantNodes()
                               .OfType<GenericNameSyntax>();

    foreach (var gen in genericTypes)
    {
        foreach (var arg in gen.TypeArgumentList.Arguments)
        {
            string typeName = arg.ToString();

            foreach (var other in fileData)
            {
                var otherFile = Path.GetFileName(other.File);

                if (otherFile == fileName)
                    continue;

                var classes = other.Root.DescendantNodes()
                                        .OfType<ClassDeclarationSyntax>();

                if (classes.Any(c => c.Identifier.Text == typeName))
                {
                    affectedFiles.Add(otherFile);
                }
            }
        }
    }
