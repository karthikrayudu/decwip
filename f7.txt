public async Task<string?> SelfCorrectBuildErrorsAsync(
    string fileContent,
    string buildErrors,
    string filePath)
{
    _output.WriteLine($"[LLM] Self-correcting: {Path.GetFileName(filePath)}...");

    // 🔒 Safety: skip if no errors
    if (string.IsNullOrWhiteSpace(buildErrors))
    {
        _output.WriteLine("[LLM-FIX] No build errors provided. Skipping fix.");
        return null;
    }

    var systemPrompt = BuildSelfCorrectionSystemPrompt();
    var userPrompt = BuildSelfCorrectionUserPrompt(fileContent, buildErrors, filePath);

    var requestBody = new
    {
        messages = new[]
        {
            new { role = "system", content = systemPrompt },
            new { role = "user", content = userPrompt }
        },
        temperature = 0.02,
        max_completion_tokens = 8192
    };

    var url = $"{_endpoint}/openai/deployments/{_deployment}/chat/completions?api-version={_apiVersion}";
    var json = JsonSerializer.Serialize(requestBody);
    var httpContent = new StringContent(json, Encoding.UTF8, "application/json");

    // 🔁 Retry logic (NEW)
    for (int attempt = 1; attempt <= MaxRetries; attempt++)
    {
        try
        {
            var response = await _httpClient.PostAsync(url, httpContent);

            if (!response.IsSuccessStatusCode)
            {
                _output.WriteLine($"[LLM-FIX] Attempt {attempt} failed: {response.StatusCode}");
                await Task.Delay(1000 * attempt);
                continue;
            }

            var responseJson = await response.Content.ReadAsStringAsync();

            using var doc = JsonDocument.Parse(responseJson);

            var content = doc.RootElement
                .GetProperty("choices")[0]
                .GetProperty("message")
                .GetProperty("content")
                .GetString();

            var cleaned = string.IsNullOrWhiteSpace(content)
                ? null
                : StripCodeFences(content);

            // 🔒 Safety check (NEW)
            if (!string.IsNullOrWhiteSpace(cleaned))
            {
                if (!cleaned.Contains("class") && !cleaned.Contains("namespace"))
                {
                    _output.WriteLine("[LLM-FIX] Suspicious output ignored.");
                    continue;
                }

                return cleaned;
            }
        }
        catch (Exception ex)
        {
            _output.WriteLine($"[LLM-FIX] Attempt {attempt} exception: {ex.Message}");
        }

        await Task.Delay(1000 * attempt);
    }

    _output.WriteLine("[LLM-FIX] All retries failed.");
    return null;
}
