// 🔥 Step 2: Accurate dependency detection
foreach (var file in allFiles)
{
    var fileName = Path.GetFileName(file.FilePath);

    // Only process files that already use the package
    if (!affectedFiles.Contains(fileName))
        continue;

    foreach (var other in allFiles)
    {
        var otherName = Path.GetFileName(other.FilePath);

        if (file.FilePath == other.FilePath)
            continue;

        // 🔍 Get all invocation expressions from source
        var source = File.ReadAllText(file.FilePath);

        // Simple detection: check if other class name is used
        foreach (var className in other.Result.ClassNames)
        {
            if (source.Contains(className + "."))
            {
                affectedFiles.Add(otherName);
            }
        }
    }
}
