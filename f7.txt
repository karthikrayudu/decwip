else
{
    Console.WriteLine(
        "[Pipeline] 📂 Full scan mode: no git diff available.");

    var allFiles = new List<(string FilePath, RoslynUsageResult Result)>();
    var affectedFiles = new HashSet<string>();

    usageReports = Directory
        .GetFiles(projectFolder, "*.cs", SearchOption.AllDirectories)
        .Select(f =>
        {
            var sourceCode = File.ReadAllText(f);
            var roslynResult = RoslynUsageAnalyzer.Analyze(sourceCode, packageNamespaces);

            // Store all files for dependency detection
            allFiles.Add((f, roslynResult));

            // ✅ Step 1: Direct package usage
            if (roslynResult.MethodNames.Any())
            {
                affectedFiles.Add(Path.GetFileName(f));
            }

            // Skip completely irrelevant files
            if (!roslynResult.ConfirmedUsingLines.Any() &&
                !roslynResult.FullyQualifiedReferenceLines.Any())
                return null;

            var report = new FileUsageReport
            {
                FilePath = f,
                WasGitChanged = true,
                OriginalContent = sourceCode
            };

            RoslynUsageAnalyzer.EnrichReport(report, roslynResult);
            return report;
        })
        .Where(r => r != null)
        .ToList()!;

    // 🔥 Step 2: Dependency detection (Helper.cs etc.)
    foreach (var file in allFiles)
    {
        var fileName = Path.GetFileName(file.FilePath);

        if (!affectedFiles.Contains(fileName))
            continue;

        foreach (var other in allFiles)
        {
            var otherName = Path.GetFileName(other.FilePath);

            if (file.FilePath == other.FilePath)
                continue;

            // If current file calls methods that belong to another file
            foreach (var method in file.Result.MethodNames)
            {
                if (other.Result.ClassNames.Contains(method) ||
                    other.Result.MethodNames.Any(m => file.Result.MethodNames.Contains(m)))
                {
                    affectedFiles.Add(otherName);
                }
            }
        }
    }

    // 🔥 Step 3: Print ONLY required files
    Console.WriteLine("\n════════════════════════════════════════");
    Console.WriteLine("  AFFECTED FILES");
    Console.WriteLine("════════════════════════════════════════");

    foreach (var file in affectedFiles)
    {
        Console.WriteLine($"- {file}");
    }

    // 🔥 Step 4: Filter usageReports
    usageReports = usageReports
        .Where(r => affectedFiles.Contains(Path.GetFileName(r.FilePath)))
        .ToList();
}
