using System;
using System.Linq;
using System.Threading.Tasks;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.MSBuild;
using Microsoft.CodeAnalysis.CSharp.Syntax;
using Microsoft.Build.Locator;

class Program
{
    static async Task Main(string[] args)
    {
        // Required for Roslyn MSBuild loading
        MSBuildLocator.RegisterDefaults();

        Console.Write("Enter .csproj path: ");
        string projectPath = Console.ReadLine();

        Console.Write("Enter Package (e.g., Newtonsoft.Json): ");
        string targetPackage = Console.ReadLine();

        var workspace = MSBuildWorkspace.Create();
        var project = await workspace.OpenProjectAsync(projectPath);
        var compilation = await project.GetCompilationAsync();

        string projectAssembly = project.AssemblyName;

        Console.WriteLine("\n🔍 Analyzing...\n");

        foreach (var tree in compilation.SyntaxTrees)
        {
            var semanticModel = compilation.GetSemanticModel(tree);
            var root = await tree.GetRootAsync();

            var classes = root.DescendantNodes().OfType<ClassDeclarationSyntax>();

            foreach (var cls in classes)
            {
                var methods = cls.DescendantNodes().OfType<MethodDeclarationSyntax>();

                foreach (var method in methods)
                {
                    bool usesTargetPackage = false;

                    var invocations = method.DescendantNodes()
                                            .OfType<InvocationExpressionSyntax>();

                    // Step 1: Check if method uses target package
                    foreach (var invocation in invocations)
                    {
                        var symbol = semanticModel.GetSymbolInfo(invocation).Symbol as IMethodSymbol;

                        if (symbol == null) continue;

                        string assembly = symbol.ContainingAssembly?.Name;

                        if (!string.IsNullOrEmpty(assembly) &&
                            assembly.Equals(targetPackage, StringComparison.OrdinalIgnoreCase))
                        {
                            usesTargetPackage = true;
                            break;
                        }
                    }

                    // Step 2: If method uses target package → print details
                    if (usesTargetPackage)
                    {
                        Console.WriteLine($"\n📄 File: {tree.FilePath}");
                        Console.WriteLine($"🏷 Class: {cls.Identifier.Text}");
                        Console.WriteLine($"🔹 Method: {method.Identifier.Text}");

                        foreach (var invocation in invocations)
                        {
                            var symbol = semanticModel.GetSymbolInfo(invocation).Symbol as IMethodSymbol;

                            if (symbol == null) continue;

                            string assemblyName = symbol.ContainingAssembly?.Name;

                            // ❌ Skip System & Microsoft base libraries
                            if (assemblyName.StartsWith("System")) continue;
                            if (assemblyName.StartsWith("Microsoft")) continue;

                            // ✅ Allow only:
                            // 1. Target package
                            // 2. Current project (internal code)
                            if (assemblyName != targetPackage && assemblyName != projectAssembly)
                                continue;

                            string methodName = symbol.Name;
                            string className = symbol.ContainingType?.Name;
                            string namespaceName = symbol.ContainingNamespace?.ToString();

                            var location = symbol.Locations.FirstOrDefault();
                            string sourceFile = location?.SourceTree?.FilePath ?? "External/Package";

                            Console.WriteLine($"\n   ➤ Call: {className}.{methodName}");
                            Console.WriteLine($"      Namespace: {namespaceName}");
                            Console.WriteLine($"      Assembly: {assemblyName}");
                            Console.WriteLine($"      Source: {sourceFile}");

                            if (assemblyName.Equals(targetPackage, StringComparison.OrdinalIgnoreCase))
                            {
                                Console.WriteLine("      🔥 Target Package");
                            }
                            else
                            {
                                Console.WriteLine("      📌 Internal Dependency");
                            }
                        }
                    }
                }
            }
        }

        Console.WriteLine("\n✅ Analysis Completed!");
    }
}// Code Generated by Sidekick is for learning and experimentation purposes only.
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp;
using Microsoft.CodeAnalysis.CSharp.Syntax;

namespace EntityAttribute_POC
{
    public static class RoslynUsageAnalyzer
    {
        /// <summary>
        /// Uses Roslyn to confirm / enrich namespace usage found by Regex.
        /// Returns a RoslynUsageResult with:
        ///   - Confirmed usings (exact line numbers from syntax tree)
        ///   - Fully-qualified type references found in method bodies
        ///   - Any syntax errors in the file
        /// </summary>
        public static RoslynUsageResult Analyze(
            string sourceCode,
            List<string> namespacesToSearch)
        {
            var result = new RoslynUsageResult();

            var syntaxTree = CSharpSyntaxTree.ParseText(sourceCode);
            var root = syntaxTree.GetCompilationUnitRoot();

            // ── 1. Syntax error detection ─────────────────────────────────────
            result.SyntaxErrors = syntaxTree
                .GetDiagnostics()
                .Where(d => d.Severity == DiagnosticSeverity.Error)
                .Select(d => $"Line {d.Location.GetLineSpan().StartLinePosition.Line + 1}: " +
                             $"{d.GetMessage()}")
                .ToList();

            // ── 2. Confirmed using directives ─────────────────────────────────
            var allUsings = root.Usings;

            foreach (var usingNode in allUsings)
            {
                var namespaceName = usingNode.Name?.ToString() ?? string.Empty;
                var lineNumber = usingNode.GetLocation()
                                            .GetLineSpan()
                                            .StartLinePosition.Line + 1;

                // Check if this using matches any of our target namespaces
                var matchedNs = namespacesToSearch.FirstOrDefault(ns =>
                    namespaceName.Equals(ns, StringComparison.OrdinalIgnoreCase) ||
                    namespaceName.StartsWith(ns + ".", StringComparison.OrdinalIgnoreCase));

                if (matchedNs != null)
                {
                    result.ConfirmedUsingLines.Add(lineNumber);
                    result.ConfirmedNamespaces.Add(matchedNs);

                    Console.WriteLine(
                        $"[Roslyn] ✅ Confirmed using: {namespaceName} " +
                        $"(line {lineNumber})");
                }
            }

            // ── 3. Fully-qualified type references (e.g. Microsoft.Data.SqlClient.SqlConnection) ──
            var qualifiedNames = root
                .DescendantNodes()
                .OfType<QualifiedNameSyntax>();

            foreach (var qn in qualifiedNames)
            {
                var fullName = qn.ToString();
                var lineNumber = qn.GetLocation()
                                   .GetLineSpan()
                                   .StartLinePosition.Line + 1;

                var matchedNs = namespacesToSearch.FirstOrDefault(ns =>
                    fullName.StartsWith(ns, StringComparison.OrdinalIgnoreCase));

                if (matchedNs != null)
                {
                    result.FullyQualifiedReferenceLines.Add(lineNumber);

                    if (!result.ConfirmedNamespaces.Contains(matchedNs))
                        result.ConfirmedNamespaces.Add(matchedNs);

                    Console.WriteLine(
                        $"[Roslyn] 🔍 Qualified ref: {fullName} " +
                        $"(line {lineNumber})");
                }
            }

            // ── 4. Class & method summary (useful for LLM context) ────────────
            result.ClassNames = root
                .DescendantNodes()
                .OfType<ClassDeclarationSyntax>()
                .Select(c => c.Identifier.Text)
                .ToList();

            result.MethodNames = root
                .DescendantNodes()
                .OfType<MethodDeclarationSyntax>()
                .Select(m => m.Identifier.Text)
                .ToList();

            return result;
        }

        /// <summary>
        /// Merges Roslyn-confirmed line numbers back into a FileUsageReport,
        /// removing false-positive Regex matches and adding any Roslyn-only finds.
        /// </summary>
        public static void EnrichReport(
            FileUsageReport report,
            RoslynUsageResult roslynResult)
        {
            // Add any Roslyn-confirmed lines not already in the report
            foreach (var line in roslynResult.ConfirmedUsingLines
                .Concat(roslynResult.FullyQualifiedReferenceLines))
            {
                if (!report.MatchedLineNumbers.Contains(line))
                    report.MatchedLineNumbers.Add(line);
            }

            // Add Roslyn-confirmed namespaces
            foreach (var ns in roslynResult.ConfirmedNamespaces)
                if (!report.MatchedNamespaces.Contains(ns))
                    report.MatchedNamespaces.Add(ns);

            // Attach syntax errors for downstream awareness
            report.SyntaxErrors = roslynResult.SyntaxErrors;
            report.ClassNames = roslynResult.ClassNames;
            report.MethodNames = roslynResult.MethodNames;

            // Re-sort lines
            report.MatchedLineNumbers = report.MatchedLineNumbers
                .OrderBy(l => l).ToList();
        }
    }

    // ── Result model returned by Roslyn analysis ─────────────────────────────
    public class RoslynUsageResult
    {
        public List<int> ConfirmedUsingLines { get; set; } = new();
        public List<int> FullyQualifiedReferenceLines { get; set; } = new();
        public List<string> ConfirmedNamespaces { get; set; } = new();
        public List<string> SyntaxErrors { get; set; } = new();
        public List<string> ClassNames { get; set; } = new();
        public List<string> MethodNames { get; set; } = new();
    }
}
