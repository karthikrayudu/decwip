// ── Step 2: Build MAIN project ─────────────────────────────────────

// 🔥 Get main project path (remove .Tests if present)
var mainProjectPath = _projectPath.Replace(".Tests", "");

Console.WriteLine("[Test] Running dotnet build (MAIN project)...");
var (buildCode, buildOut, buildErr) = await RunDotnetAsync(
    $"build \"{mainProjectPath}\" --no-restore --verbosity normal");

result.BuildOutput = buildOut + buildErr;
result.BuildSucceeded = buildCode == 0;

if (!result.BuildSucceeded)
{
    Console.WriteLine("[Test] ❌ Main project build failed.");
    result.BuildErrors = buildErr;
    result.ParsedBuildErrors = ParseBuildErrors(buildOut + buildErr);
    return result;
}

Console.WriteLine("[Test] ✅ Main project build succeeded.");
