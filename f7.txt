// 🔥 Add dependent files (User.cs etc.)
foreach (var report in usageReports)
{
    if (report.ClassNames == null) continue;

    foreach (var className in report.ClassNames)
    {
        var dependentFile = Directory
            .GetFiles(projectFolder, "*.cs", SearchOption.AllDirectories)
            .FirstOrDefault(f =>
                File.ReadAllText(f).Contains($"class {className}")
            );

        if (dependentFile != null)
        {
            var fileName = Path.GetFileName(dependentFile);
            if (!affectedFiles.Contains(fileName))
            {
                affectedFiles.Add(fileName);
            }
        }
    }
}
