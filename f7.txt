        public static LLMCodeMigrator? TryCreate(EventData data)
        {
            // var endpoint = data.llmEndpoint ?? Environment.GetEnvironmentVariable("AZURE_OPENAI_ENDPOINT");
            // var deployment = data.llmDeployment ?? Environment.GetEnvironmentVariable("AZURE_OPENAI_DEPLOYMENT") ?? "gpt-4o";
            // var apiKey = data.llmApiKey ?? Environment.GetEnvironmentVariable("AZURE_OPENAI_API_KEY");
            var endpoint = !string.IsNullOrWhiteSpace(data.llmEndpoint)? data.llmEndpoint: Environment.GetEnvironmentVariable("AZURE_OPENAI_ENDPOINT");
            var deployment = !string.IsNullOrWhiteSpace(data.llmDeployment)? data.llmDeployment: Environment.GetEnvironmentVariable("AZURE_OPENAI_DEPLOYMENT");
            var apiKey = !string.IsNullOrWhiteSpace(data.llmApiKey)? data.llmApiKey: Environment.GetEnvironmentVariable("AZURE_OPENAI_API_KEY");

            if (string.IsNullOrWhiteSpace(endpoint) || string.IsNullOrWhiteSpace(apiKey))
            {
                Console.WriteLine("[LLM] ⚠ Azure OpenAI not configured. Set AZURE_OPENAI_ENDPOINT + AZURE_OPENAI_API_KEY, or add llmEndpoint/llmApiKey to event.json.");
                return null;
            }

            return new LLMCodeMigrator(endpoint, deployment!, apiKey!);
        }

        public async Task<List<FileMigrationResult>> MigrateFilesAsync(
            List<FileUsageReport> usageReports,
            string packageName,
            string fromVersion,
            string toVersion)
        {
            var results = new List<FileMigrationResult>();

            foreach (var report in usageReports)
            {
                // Console.WriteLine($"[LLM] Migrating: {Path.GetFileName(report.FilePath)}...");//error
                   _output.WriteLine($"[LLM] Migrating: {Path.GetFileName(report.FilePath)}...");//error

                var result = await MigrateSingleFileWithRetryAsync(
                    report, packageName, fromVersion, toVersion);

                results.Add(result);

                if (result.MigrationSucceeded && !result.RequiresManualReview)
                {
                    // Console.WriteLine($"[LLM] ✅ Done: {Path.GetFileName(report.FilePath)}");
                    _output.WriteLine($"[LLM] ✅ Done: {Path.GetFileName(report.FilePath)}");
                    foreach (var c in result.ChangesApplied)
                    {
                        // Console.WriteLine($"       • {c}");
                        _output.WriteLine($"       • {c}");
                    }
                }
                else if (result.RequiresManualReview)
                {
                    // Console.WriteLine($"[LLM] ⚠ Manual review: {result.ManualReviewReason}");
                    _output.WriteLine($"[LLM] ⚠ Manual review: {result.ManualReviewReason}");
                }
                else
                {
                    // Console.WriteLine($"[LLM] ❌ Failed: {result.ErrorMessage}");
                    _output.WriteLine($"[LLM] ❌ Failed: {result.ErrorMessage}");
                }
            }

            return results;
        }

        private async Task<FileMigrationResult> MigrateSingleFileWithRetryAsync(
            FileUsageReport report,
            string packageName,
            string fromVersion,
            string toVersion)
        {
            FileMigrationResult? lastResult = null;

            for (int attempt = 1; attempt <= MaxRetries; attempt++)
            {
                try
                {
                    lastResult = await MigrateSingleFileAsync(
                        report, packageName, fromVersion, toVersion);

                    if (lastResult.MigrationSucceeded)
                    {
                        return lastResult;
                    }

                    // Console.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} failed. Retrying...");
                    _output.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} failed. " +
                             $"Reason: {lastResult.ErrorMessage}");
                    //_output.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} failed. Retrying...");
                    await Task.Delay(TimeSpan.FromSeconds(attempt * 2));
                }
                catch (Exception ex)
                {
                    // Console.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} threw: {ex.Message}");
                    var fullError = ex.InnerException != null
                ? $"{ex.Message} → {ex.InnerException.Message}"
                : ex.Message;
                    _output.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} threw: {ex.Message}");
                    //_output.WriteLine($"[LLM] Attempt {attempt}/{MaxRetries} threw: {fullError}");
                    //_output.WriteLine($"[LLM] Exception type: {ex.GetType().FullName}");
                    await Task.Delay(TimeSpan.FromSeconds(attempt * 2));

                    lastResult = new FileMigrationResult
                    {
                        FilePath = report.FilePath,
                        MigrationSucceeded = false,
                        ErrorMessage = ex.Message
                    };
                }
            }

            return lastResult ?? new FileMigrationResult
            {
                FilePath = report.FilePath,
                MigrationSucceeded = false,
                ErrorMessage = "All retry attempts exhausted."
            };
        }

        private async Task<FileMigrationResult> MigrateSingleFileAsync(
            FileUsageReport report,
            string packageName,
            string fromVersion,
            string toVersion)
        {
            var namespacesUsed = string.Join(", ", report.MatchedNamespaces);
            var affectedLineNums = string.Join(", ", report.MatchedLineNumbers);
            var fileName = Path.GetFileName(report.FilePath);

            var systemPrompt = BuildMigrationSystemPrompt(packageName, fromVersion, toVersion);
            var userPrompt = BuildMigrationUserPrompt(
                fileName,
                packageName,
                fromVersion,
                toVersion,
                namespacesUsed,
                affectedLineNums,
                report.OriginalContent);

            var requestBody = new
            {
                messages = new[]
                {
                    new { role = "system", content = systemPrompt },
                    new { role = "user", content = userPrompt }
                },
                temperature = 0.05,
                max_completion_tokens = 8192,
                response_format = new { type = "json_object" }
            };

            var url = $"{_endpoint}/openai/deployments/{_deployment}/chat/completions?api-version={_apiVersion}";
            var json = JsonSerializer.Serialize(requestBody);
            var httpContent = new StringContent(json, Encoding.UTF8, "application/json");

            var response = await _httpClient.PostAsync(url, httpContent);

            if (!response.IsSuccessStatusCode)
            {
                var errorBody = await response.Content.ReadAsStringAsync();
                return new FileMigrationResult
                {
                    FilePath = report.FilePath,
                    MigrationSucceeded = false,
                    ErrorMessage = $"HTTP {(int)response.StatusCode}: {errorBody}"
                };
            }

            var responseJson = await response.Content.ReadAsStringAsync();
            return ParseLLMResponse(responseJson, report.FilePath);
        }

        public async Task<string?> SelfCorrectBuildErrorsAsync(
            string fileContent,
            string buildErrors,
            string filePath)
        {
            // Console.WriteLine($"[LLM] Self-correcting: {Path.GetFileName(filePath)}...");
            _output.WriteLine($"[LLM] Self-correcting: {Path.GetFileName(filePath)}...");

            var systemPrompt = BuildSelfCorrectionSystemPrompt();
            var userPrompt = BuildSelfCorrectionUserPrompt(fileContent, buildErrors, filePath);

            var requestBody = new
            {
                messages = new[]
                {
                    new { role = "system", content = systemPrompt },
                    new { role = "user", content = userPrompt }
                },
                temperature = 0.02,
                max_completion_tokens = 8192
            };

            var url = $"{_endpoint}/openai/deployments/{_deployment}/chat/completions?api-version={_apiVersion}";
            var json = JsonSerializer.Serialize(requestBody);
            var httpContent = new StringContent(json, Encoding.UTF8, "application/json");

            var response = await _httpClient.PostAsync(url, httpContent);
            if (!response.IsSuccessStatusCode)
            {
                return null;
            }

            var responseJson = await response.Content.ReadAsStringAsync();

            using var doc = JsonDocument.Parse(responseJson);
            var content = doc.RootElement
                .GetProperty("choices")[0]
                .GetProperty("message")
                .GetProperty("content")
                .GetString();

            return string.IsNullOrWhiteSpace(content) ? null : StripCodeFences(content);
        }

        private static string StripCodeFences(string raw)
        {
            var lines = raw.Split('\n').ToList();

            if (lines.Count > 0 && lines[0].TrimStart().StartsWith("```", StringComparison.Ordinal))
            {
                lines.RemoveAt(0);
            }

            if (lines.Count > 0 && lines[^1].Trim() == "```")
            {
                lines.RemoveAt(lines.Count - 1);
            }

            return string.Join('\n', lines).Trim();
        }

        private static FileMigrationResult ParseLLMResponse(
            string responseJson,
            string filePath)
        {
            try
            {
                using var outerDoc = JsonDocument.Parse(responseJson);
                var content = outerDoc.RootElement
                    .GetProperty("choices")[0]
                    .GetProperty("message")
                    .GetProperty("content")
                    .GetString() ?? string.Empty;

                using var innerDoc = JsonDocument.Parse(content);
                var root = innerDoc.RootElement;

                var migratedCode = root.TryGetProperty("migratedCode", out var mc)
                    ? mc.GetString() ?? string.Empty
                    : string.Empty;

                var requireReview = root.TryGetProperty("requiresManualReview", out var rev)
                    && rev.GetBoolean();

                var reviewReason = root.TryGetProperty("manualReviewReason", out var rr)
                    ? rr.GetString() ?? string.Empty
                    : string.Empty;

                var changes = new List<string>();
                if (root.TryGetProperty("changesApplied", out var changesEl) &&
                    changesEl.ValueKind == JsonValueKind.Array)
                {
                    foreach (var c in changesEl.EnumerateArray())
                    {
                        var s = c.GetString();
                        if (!string.IsNullOrWhiteSpace(s))
                        {
                            changes.Add(s);
                        }
                    }
                }

                return new FileMigrationResult
                {
                    FilePath = filePath,
                    MigratedContent = migratedCode,
                    ChangesApplied = changes,
                    RequiresManualReview = requireReview,
                    ManualReviewReason = reviewReason,
                    MigrationSucceeded = !string.IsNullOrWhiteSpace(migratedCode)
                };
            }
            catch (Exception ex)
            {
                return new FileMigrationResult
                {
                    FilePath = filePath,
                    MigrationSucceeded = false,
                    ErrorMessage = $"Failed to parse LLM response: {ex.Message}. Raw: {responseJson[..Math.Min(300, responseJson.Length)]}"
                };
            }
        }
        public async Task<FileMigrationResult> MigrateLargeFileBySnippetAsync(
                FileUsageReport report,
                string packageName,
                string fromVersion,
                string toVersion)
            {
                var originalContent = report.OriginalContent ?? string.Empty;

                // Build a focused snippet from matched lines
                var lines = originalContent.Split('\n');
                var snippetLines = report.MatchedLineNumbers
                    .Where(n => n >= 1 && n <= lines.Length)
                    .SelectMany(n => new[]
                    {
                        Math.Max(0, n - 3),    // 3 lines before
                        n - 1,                  // the matched line (0-indexed)
                        Math.Min(lines.Length - 1, n + 2)  // 2 lines after
                    })
                    .Distinct()
                    .OrderBy(i => i)
                    .Select(i => lines[i]);

                var snippet = string.Join("\n", snippetLines);

                // Ask LLM to migrate just the snippet
                var systemPrompt = BuildMigrationSystemPrompt(packageName, fromVersion, toVersion);
                var userPrompt = $"Migrate only this snippet from {packageName} {fromVersion} to {toVersion}.\n\nSNIPPET:\n{snippet}\n\nReturn JSON with: updatedSnippet, changesApplied, requiresManualReview, manualReviewReason";

                var requestBody = new
                {
                    messages = new[]
                    {
                        new { role = "system", content = systemPrompt },
                        new { role = "user", content = userPrompt }
                    },
                    temperature = 0.05,
                    max_completion_tokens = 4096,
                    response_format = new { type = "json_object" }
                };

                var url = $"{_endpoint}/openai/deployments/{_deployment}/chat/completions?api-version={_apiVersion}";
                var json = System.Text.Json.JsonSerializer.Serialize(requestBody);
                var httpContent = new StringContent(json, Encoding.UTF8, "application/json");

                var response = await _httpClient.PostAsync(url, httpContent);
                if (!response.IsSuccessStatusCode)
                {
                    return new FileMigrationResult
                    {
                        FilePath = report.FilePath,
                        MigrationSucceeded = false,
                        ErrorMessage = $"HTTP {(int)response.StatusCode}"
                    };
                }

                var responseJson = await response.Content.ReadAsStringAsync();

                try
                {
                    using var outerDoc = System.Text.Json.JsonDocument.Parse(responseJson);
                    var content = outerDoc.RootElement
                        .GetProperty("choices")[0]
                        .GetProperty("message")
                        .GetProperty("content")
                        .GetString() ?? string.Empty;

                    using var innerDoc = System.Text.Json.JsonDocument.Parse(content);
                    var root = innerDoc.RootElement;

                    var updatedSnippet = root.TryGetProperty("updatedSnippet", out var us)
                        ? us.GetString() ?? string.Empty
                        : string.Empty;

                    // Replace the original snippet with the updated snippet in the full file
                    var migratedContent = string.IsNullOrWhiteSpace(updatedSnippet)
                        ? originalContent
                        : originalContent.Replace(snippet, updatedSnippet);

                    var changes = new List<string>();
                    if (root.TryGetProperty("changesApplied", out var changesEl) &&
                        changesEl.ValueKind == System.Text.Json.JsonValueKind.Array)
                    {
                        foreach (var c in changesEl.EnumerateArray())
                        {
                            var s = c.GetString();
                            if (!string.IsNullOrWhiteSpace(s)) changes.Add(s);
                        }
                    }

                    return new FileMigrationResult
                    {
                        FilePath = report.FilePath,
                        MigratedContent = migratedContent,
                        ChangesApplied = changes,
                        MigrationSucceeded = !string.IsNullOrWhiteSpace(updatedSnippet),
                        RequiresManualReview = root.TryGetProperty("requiresManualReview", out var rev) && rev.GetBoolean(),
                        ManualReviewReason = root.TryGetProperty("manualReviewReason", out var rr) ? rr.GetString() ?? string.Empty : string.Empty
                    };
                }
                catch (Exception ex)
                {
                    return new FileMigrationResult
                    {
                        FilePath = report.FilePath,
                        MigrationSucceeded = false,
                        ErrorMessage = $"Snippet parse failed: {ex.Message}"
                    };
                }
            }
