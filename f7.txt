{
    Console.WriteLine("[Fix] Attempting to fix test failures...");

    foreach (var file in Directory.GetFiles(_projectFolder, "*.cs", SearchOption.AllDirectories))
    {
        var code = File.ReadAllText(file);

        var fixedCode = await _migrator.SelfCorrectBuildErrorsAsync(

            testErr,

        if (!string.IsNullOrWhiteSpace(fixedCode))
        {

        }
    }

    Console.WriteLine("[Fix] Re-running tests...");
    (testCode, testOut, testErr) =
        await RunDotnetAsync($"test \"{testFolder}\"");
